Current page url

echo $GLOBALS['TSFE']->page['uid'];

echo $GLOBALS['TSFE']->cObj->getTypoLink_URL(12345678);

Typoscript

echo $GLOBALS['TSFE']->tmpl->setup['somethingtoaccess.']['aproperty'];
